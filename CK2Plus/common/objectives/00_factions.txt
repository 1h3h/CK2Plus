# Feudal faction: Desires low CA, low feudal taxes and obligations
faction_feudal_emperor = {
	type = character
	
	# Plotter scope
	potential = {	
		ai = yes	
		prisoner = no
		is_ruler = yes
		is_feudal = yes 
		independent = no
		liege = { NOT = { character = ROOT } }
		liege = { tier = emperor }
		NOT = { liege = { has_character_flag = faction_feudal_war } }
		age = 16
		NOT = { trait = incapable }

			liege = {
				any_vassal = {
					leads_faction = faction_feudal_emperor
					NOT = { character = ROOT }
				}
			}
		}
		NOT = { liege = { in_faction = faction_feudal_emperor } }

		factor = 100
		modifier = {
			factor = 10
			tier = king
		}
		modifier = {
			factor = 0.1
			tier = count
		}			
		modifier = {
			factor = 0
			ROOT = { NOT = { has_character_flag = faction_feudal_supporter } }
		}
		modifier = {
			factor = 0
			ROOT = { 
				liege = {
					current_heir = {
						character = ROOT
					}
				}
			}
		}		
	}
	
	# AI membership weight: ROOT is the prospective member. FROM is the faction leader. FROMFROM is the target title or character.
	membership = {
		factor = 100
		modifier = {
			factor = 0
			NOT = { has_character_flag = faction_feudal_supporter }
		}
		modifier = {
			factor = 0
			prisoner = yes
		}		
		modifier = {
			factor = 0
			OR = {
				trait = incapable

			}
		}

	success = {	
		always = no
