#Send mission to neighbour pagan
character_event = {
	id = 1200000
	picture = GFX_evt_heretic
	border = "GFX_event_normal_frame_religion"
	
	only_rulers = yes
	capable_only = yes
	min_age = 16
	
	trigger = {
		war = no
		OR = {
			religion_group = christian
			religion_group = muslim
		}
		NOT = { trait = cynical }
		OR = {
			NOT = { has_character_flag = missionary_expedition }
			had_character_flag = {
				flag = missionary_expedition
				days = 3650
			}
		}
		any_independent_ruler = {
			war = no
			OR = {
				religion_group = pagan_group
				religion_group = african_pagan
				religion_group = eastern_pagan
			}
			opinion = {
				who = ROOT
				value = -50
			}			
			NOT = { trait = zealous }
			NOT = { trait = incapable }
			age = 16
			NOT = { age = 45 }
			NOT = { has_character_flag = missionary_expedition }
			NOT = { higher_tier_than = ROOT }
			NOT = { tier = emperor }
			any_realm_province = {
				any_neighbor_province = {
					owner = {
						OR = {
							same_realm = ROOT
							character = ROOT
						}
					}
				}
			}		
		}
	}
	
	mean_time_to_happen = {
		years = 50
		modifier = {
			factor = 5
			tier = count
		}
		modifier = {
			factor = 3
			tier = duke
		}
		modifier = {
			factor = 1.5
			tier = king
		}		
		modifier = {
			factor = 1.1
			trait = slothful
		}	
		modifier = {
			factor = 0.9
			trait = diligent
		}			
		modifier = {
			factor = 0.75
			trait = zealous
		}
		modifier = {
			factor = 0.5
			piety = 100
		}		
		modifier = {
			factor = 0.5
			piety = 250
		}
		modifier = {
			factor = 0.5
			piety = 500
		}
		modifier = {
			factor = 0.5
			piety = 750
		}
		modifier = {
			factor = 0.5
			piety = 1000
		}
	}
		
	desc = "EVTDESC1200000"
	
	option = {
		name = "EVTOPTA1200000"
		ai_chance = {
			factor = 100
		}
		wealth = -100
		if = {
			limit = { NOT = { trait = zealous } }
			random = {
				chance = 33
				add_trait = zealous
				hidden_tooltip = {
					character_event = {
						id = 38269 #Notify Zealous
					}
				}
			}
		}		
		random_independent_ruler = {
			limit = {
				war = no
				OR = {
					religion_group = pagan_group
					religion_group = african_pagan
					religion_group = eastern_pagan
				}
				opinion = {
					who = ROOT
					value = -50
				}	
				NOT = { trait = zealous }
				NOT = { trait = zealous }
				NOT = { trait = incapable }
				age = 16
				NOT = { age = 45 }
				NOT = { has_character_flag = missionary_expedition }
				NOT = { higher_tier_than = ROOT }
				NOT = { tier = emperor }
				any_realm_province = {
					any_neighbor_province = {
						owner = {
							OR = {
								same_realm = ROOT
								character = ROOT
							}
						}
					}
				}	
			}
			set_character_flag = missionary_expedition
			character_event = { id = 1200001 days = 14 tooltip = EVTTOOLTIP1200001 }
		}
		set_character_flag = missionary_expedition
	}
	option = {
		name = "EVTOPTB1200000"	
		ai_chance = {
			factor = 10
			modifier = {
				factor = 5
				NOT = { wealth = 200 }
			}
		}
		piety = -50
		if = {
			limit = { trait = zealous }
			remove_trait = zealous
		}
		set_character_flag = missionary_expedition
	}	
}
